FooBar

This ruby program is an adaptation of a classic technical interview problem which displays sequences of the Foobar pattern. 


      def create_array(n)   
        values = []
        new_value = 1
        until values.length == n
          values.push(new_value)
          new_value += 1
        end
        return values
      end

      puts "How many items do you want to see? "
      value_array = create_array(gets.to_i)
      puts ""

      def foobar(array)
        array.each do |n|
          if(n % 3 == 0 && n % 5 == 0)
            puts "FooBar"
          elsif (n % 3 == 0)
            puts "Foo"
          elsif (n % 5 == 0)
            puts "Bar"
          else
            puts n
          end
        end
      end
        
      foobar(value_array)
      puts ""
